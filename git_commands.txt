check status
- git status

add elements to commit
- git add *

make commit
- git commit -m "message"

make push
- git push origin master

check recent commits
- git log
- git log --pretty=oneline
- git log --pretty=oneline -2

pull changes
- git pull

set name and email
- git config --global user.name "First Last Name"
- git config --global user.email "email@example.com"

check name and email
- git config user.name
- git config user.email

edit file from the Git Bash
- vi filename

commands for the vi editor
- to start editing, type 'i'. This makes the editor go into Insert mode
- when finished editing, press Esc to get out of editor mode
- to save the file, type ':w'
- to exit the file, type ':q'
- to exit the file without saving, type ':q!'
